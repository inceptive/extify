/**
 * @class Ext.Ajax
 *
 * 

 * Shortcut class for Ext.data.Connection. 
 * 

 * 
 * @see Ext.data.Connection  
 */
Ext.define("Ext.Ajax", { 

    
    singleton: true,
    
    
    
/**
     * Fires a request via HTTP protocol as configured in the given config
     * object: Works like Ext JS's Ext.Ajax / Ext.data.Connection impl.          
     *     
     * 

     * Possible configuration:
     * 

     *
     * 

     *  
     * 
     * Ext.Ajax.request({
     *     url: '',
     *     success: function(result) {},
     *     failure: function() {},
     *     headers: { // Special HTTP request headers like X-*
     *         X-Requested-With: 'Extanium'
     *     },
     *     params: { // HTTP POST parameters
     *         user: 'foo',
     *         password: 'bar'
     *     }, // OR INSTEAD OF params!! data: 'ABC' for plain data sending
     *     timeout: 5, // In seconds
     *     method: 'POST', // Which HTTP method to use?
     *     json: true, // Parses incomming text-result automatically and provide result.json object
     *     async: true, 
     *     secure: false // HTTPS cert checking
     * });
     * 

     *     
     * 

     * 
     * 

     * The example above shows the default configuration.
     * 

     *
     * 

     * 
     * success: function(result) {
     *     debug(result.data);
     *     debug(result.xml);
     *     debug(result.text);
     *     debug(result.json); // Only if json: true is defined (provides auto-deserialization)
     * } 
     * 

     * 
          
     * 
     * @param {Object} cfg Configuration of the request
     * @return {Ti.Network.HttpClient|Boolean} HTTP client instance or false          
     */         
    request: function(cfg) {
        return new Ext.net.HttpClient().request(cfg);
    }
});
