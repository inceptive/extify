/**
 * @class Ext.Application
 *  
 * Represents the application. 
 * Instance registers as: App in global scope!   
 * 
 * To be used via: Ext.application({
 *     // Configuration goes here
 * })  
 */ 
Ext.define('Ext.Application', {

    mixins: ['Ext.util.Observable'],

    statics: {
    
        
/**
         * Defines an application and registers it's controllers generically. 
         * 
         * @param {Object} cfg Application configuration
         * @return void     
         */         
        create: function(cfg) {
            global.App = new Ext.Application(cfg);
        }
    },


    
/**
     * @cfg {String} name Name of the app
     */         
    name: '',
    
    
    
/**
     * @cfg {String} appFolder Application folder
     */
    appFolder: 'app',
    
    
    
/**
     * @cfg {Array} controllers Controller names
     */         
    controllers: [],
    
    
    
/**
     * Constructor for the application class
     *
     * @param {Object} cfg Config overlay
     * @return void          
     */         
    constructor: function(cfg) {
    
        // Overlay all given Ext.application()  
        // configs to this instance
        Ext.apply(this, cfg);
        
        // Load controllers
        Ext.loadClassesByNames(this.controllers, this.appFolder + "/controller");
        
        // Call the onLaunch() method
        this.onLaunch();
    },
    
    
    
/**
     * Get's automatically called on app launch.
     * @return void     
     */         
    onLaunch: function() {}
});

// Shortcut
Ext.application = Ext.Application.create;
