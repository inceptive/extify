/**
 * @class Ext.View
 *  
 * Represents a view.   
 * 
 *
 * To be used via: Ext.view({
 *     // Configuration goes here
 * })  
 *    
 * Each instance registers by it's name attribute and is
 * fetchable by:
 * 
 * var view = Ext.getView('nameOfTheView');  
 */ 
Ext.define('Ext.View', {
                            
    mixins: ['Ext.util.Observable'],
    extend: 'Ext.Component',

    statics: {
    
        
/**
         * Defines an view. 
         * 
         * @param {Object} cfg View configuration
         * @return void     
         */         
        create: function(cfg) {
        
            if (typeof global.Views == "undefined") {
                global.Views = {};
            };
            global.Views[cfg.name] = new Ext.View(cfg);
        },
        
        
        
/**
         * Returns the view instance for a name
         * 
         * @param {String} name Name of the view                  
         * @return {Ext.View}         
         */                 
        get: function(name) {
            return global.Views[name];
        }
    },


    
/**
     * @cfg {String} Name of the view
     */         
    name: '',
    
    
    
/**
     * Constructor for the view class
     *
     * @param {Object} cfg Config overlay
     * @return void          
     */         
    constructor: function(cfg) {
    
        // Call Component constructor for dynamic overlaying
        Ext.View.superclass.constructor.apply(this, arguments);
        
        // Call the init() method
        this.init();
    },
    
    
    
/**
     * Get's automatically called on instance creation.
     * @return void     
     */         
    init: function() {}
});

// Shortcut
Ext.view = Ext.View.create;
Ext.getView = Ext.View.get;
